import subprocess

from docx import Document
from docx.shared import Inches, Pt, Cm

def set_column_break(paragraph):
    """Вставляет разрыв колонки."""
    run = paragraph.add_run()
    br = OxmlElement('w:br')
    br.set(qn('w:type'), 'column')
    run._element.append(br)

def apply_bullet_style(paragraph):
    """Применяет стиль для буквицы (имитация)."""
    # Увеличиваем первую букву через run
    if paragraph.text:
        first_char = paragraph.text[0]
        rest_text = paragraph.text[1:]
        paragraph.clear()
        run_first = paragraph.add_run(first_char)
        run_first.font.size = Pt(36)
        run_first.font.bold = True
        run_rest = paragraph.add_run(rest_text)
